Ejecución de un bot

Un bot no es un programa que ejecutemos y finalice, realizando una acción concreta.
Es un programa que responde a llamadas que no sabemos cuando se van a producir, por tanto debe tener un sistema de espera síncrona o asíncrona.

Ejecución de un bot

Usaremos polling, que explicado de una manera sencilla, se dedica a preguntar repetidas veces al programa si hay algo que tratar, hasta que suceda algún evento y pueda actuar.

Por tanto debemos añadir esto al final del programa. 


						bot.polling()

Solución propuesta


from queue import Queue

photos = Queue()

@bot.message_handler(content_types=['photo'])
def store_photo(message):
    photos.put(message.photo[1].file_id)


@bot.message_handler(commands=['get'])
def get_photo(message):
    photo_id = photos.get_nowait()
    bot.send_photo(message.chat.id , photo_id)

from queue import Queue

lista = {}

@bot.message_handler(content_types=['photo'])
def store_photo(message):
    if not message.from_user.id in lista :
        photos = Queue()
        lista[message.from_user.id] = photos
    lista[message.from_user.id].put(message.photo[1].file_id)


@bot.message_handler(commands=["get"])
def get_photo(message):
    if message.from_user.id in lista :
        photo_id = lista[message.from_user.id].get()
        bot.send_photo(message.chat.id , photo_id)
    else:
        bot.send_message(message.chat.id, "No tienes nada en la cola")

Un ejemplo concreto sobre botones

Preparando el terreno

Creamos unas variables globales para almacenar y mantener los datos 


markup_sumas = types.InlineKeyboardMarkup()
sumatorio = 0

Creamos la función con el teclado


@bot.message_handler(commands=['sumar'])
def test_sumar(self):
	text = 'Número acumulado: 0'
	global markup_sumas
	markup_sumas = types.InlineKeyboardMarkup()

	#Primera fila de la botonera
	a=[]
	a.append(types.InlineKeyboardButton('7' ,callback_data='7'))
	a.append(types.InlineKeyboardButton('8' ,callback_data='8'))
	a.append(types.InlineKeyboardButton('9' ,callback_data='9'))

	#Segunda fila de la botonera
	b=[]
	b.append(types.InlineKeyboardButton('4' ,callback_data='4'))
	b.append(types.InlineKeyboardButton('5' ,callback_data='5'))
	b.append(types.InlineKeyboardButton('6' ,callback_data='6'))

	#Tercera fila de la botonera
	c=[]
	c.append(types.InlineKeyboardButton('1' ,callback_data='1'))
	c.append(types.InlineKeyboardButton('2' ,callback_data='2'))
	c.append(types.InlineKeyboardButton('3' ,callback_data='3'))

	#Última fila de la botonera
	d=[]
	d.append(types.InlineKeyboardButton('Listo' ,callback_data="sumar"))

	#Añadimos las filas creadas al teclado definido
	markup_sumas.row(*a)
	markup_sumas.row(*b)
	markup_sumas.row(*c)
	markup_sumas.row(*d)

	#Devolvemos el mensaje
	bot.reply_to(self, text, reply_markup=markup_sumas)

Creamos la funcion

La funcion detecta el valor pulsado y lo trata según convenga 


@bot.callback_query_handler(func=lambda call: '1'or'2'or'3'or'4'or'5'or'6'or'7'or'8'or'9'or'sumar' in call.data)
def get_mssg_numero(call):
        global sumatorio
        if(call.data == 'sumar'):
            bot.edit_message_text("Este es el número final: "+str(sumatorio), call.from_user.id, call.message.message_id)
        else:
            sumatorio += int(call.data)
            bot.edit_message_text("Número acumulado: "+str(sumatorio), call.from_user.id, call.message.message_id,reply_markup=markup_sumas)
